class Solution {
public:
    void moveZeroes(vector<int>& nums) {
      int   n=nums.size();
      if (n>1){
      int   j=-1;
        for (int i=0;i<n;i++){
            if (nums[i]==0){
                j=i;
                break;
                }
            }
        if(j!=-1){    
        for (int k =j+1;k<n;k++){
            if(nums[k]!=0){
                swap(nums[j],nums[k]);
                j++;

            }  }         
        
        }          




        }
    }
    
};
