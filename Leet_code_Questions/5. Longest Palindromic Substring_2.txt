#include <string>
#include <vector>

using namespace std;

class Solution {
public:
    string longestPalindrome(string s) {
        int n = s.length();
        vector<vector<bool>> dp(n, vector<bool>(n, false));

        int start = 0; 
                int maxLen = 1; 

      
        for (int i = 0; i < n; i++) {
            dp[i][i] = true;
        }

             for (int i = 0; i < n - 1; i++) {
            if (s[i] == s[i + 1]) {
                dp[i][i + 1] = true;
                start = i;
                maxLen = 2;
            }
        }

       
        for (int len = 3; len <= n; len++) {
            for (int i = 0; i <= n - len; i++) {
                int j = i + len - 1;

               
                if (dp[i + 1][j - 1] && s[i] == s[j]) {
                    dp[i][j] = true;

                   
                    if (len > maxLen) {
                        start = i;
                        maxLen = len;
                    }
                }
            }
        }

        
        return s.substr(start, maxLen);
    }
};
